```C++

int divide(int a, int b) {
    if (b == 0) {
        throw string("Division by zero");
    } else {
        return a / b;
    }
}

int main() {
    try {
        divide(2, 0);
    } catch (string &e)  {
        cout << e << endl;
    }
    return 0;
}
```
